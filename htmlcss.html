  • Boucles
// while
let i = 0;
while (i < 3) {
  console.log(i);
  i++;
}

// do while
let j = 0;
do {
  console.log(j);
  j++;
} while (j < 3);

// for
for (let k = 0; k < 3; k++) {
  console.log(k);
}
