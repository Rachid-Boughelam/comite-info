Premier projet Java : Le Jeu du Plus ou du Moins

1. Créer un projet Java (avec IntelliJ)

package org.rachidboughelam;

                    import java.util.Random;
                    import java.util.Scanner;

                    public class Main {
                        public static void main(String[] args) {
                            while (true) {
                                Random rand = new Random();
                                int reponse = rand.nextInt(10) + 1;
                                int nombreSaisi = 0;

                                System.out.println("Une partie ? 1 pour oui, 2 pour non");
                                Scanner scanner1 = new Scanner(System.in);
                                int autrePartie = scanner1.nextInt();

                                if (autrePartie == 1) {
                                    System.out.println("Tu as choisi oui");
                                } else if (autrePartie == 2) {
                                    System.out.println("Tu as choisi non");
                                    System.exit(0);
                                } else {
                                    System.out.println("Erreur : tu dois taper 1 pour oui ou 2 pour non.");
                                    System.exit(0);
                                }

                                System.out.println("Trouve le nombre mystère !");
                                while (reponse != nombreSaisi) {
                                    Scanner scanner2 = new Scanner(System.in);
                                    nombreSaisi = scanner2.nextInt();

                                    if (nombreSaisi < reponse) {
                                        System.out.println("C'est plus !");
                                    } else if (nombreSaisi > reponse) {
                                        System.out.println("C'est moins !");
                                    } else {
                                        System.out.println("Bravo ! Tu as gagné");
                                    }
                                }
                            }
                        }
                    }
