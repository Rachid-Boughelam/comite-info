Notre programme

  • Déclaration de variables avec let, var et const
    • 

        const pays = "Canada"; //ne change jamais
        var nom = "Alice";
        let age = 30;
        console.log("Bonjour, je m'appelle", nom , "je vis au" , pays , "et j'ai", age ,"ans."); 
        // affichera Bonjour, je m'appelle Alice je vis au Canada et j'ai 30 ans.
  • Les conditions avec if et else
    • 

        let temperature = 25;

        if (temperature > 30) {
        console.log("Il fait très chaud !");
        } else if (temperature > 20) {
        console.log("Il fait bon.");
        } else {
        console.log("Il fait frais.");
        }
        // affichera il fait bon
  • Les boucles avec while, do while et for
    • 

        // while
        let i = 0;
        while (i < 3) {
        console.log(i);
        i++;
        }
        // affichera 0 1 2

        // do while
        let j = 0;
        do {
        console.log(j);
        j++;
        } while (j < 3);
        // affichera 0 1 2

        // for
        for (let k = 0; k < 3; k++) {
        console.log(k);
        }
        // affichera 0 1 2
